// Hard cap — überall gleich anwenden
cap(score, min=0, max=25): 
  Math.min(max, Math.max(min, score))

// Kakao-Prozent defensiv lesen
getCacaoPercent(product):
  1. product.product_name regex: 
     /(\d+)\s*%.*kakao|kakao.*?(\d+)\s*%/i
  2. product.ingredients_text regex:
     /kakaomasse.*?(\d+)|kakaoanteil.*?(\d+)|
      davon kakao.*?(\d+)/i
  3. return null wenn nicht gefunden

// Zucker-Kontext (nicht binär!)
getSugarContext(ingredients_text): SugarContext
  hasNaturalSugar: 
    /rohrzucker|kokosblütenzucker|
     dattelsirup|ahornsirup/i
  hasIndustrialSugar:
    /\bzucker\b|glukosesirup|fruktosesirup|
     maissirup|maltodextrin/i
  hasFruitSugar:
    /fruchtsaft|fruchtpüree|apfelsaft|
     fruchtzucker(?!sirup)/i

// Fluorid-Check
hasFluoride(ingredients_text):
  /natriumfluorid|sodium fluoride|
   fluoridiertes wasser/i

## D) PROFIL-FIT (0-25):
start = 25, penalty = 0
sugar = product.nutriments.sugars_100g ?? 0
sugarCtx = getSugarContext(ingredients_text)
cacao = getCacaoPercent(product) ?? 0
warningFlags = []

// Fluorid immer prüfen
if hasFluoride(ingredients_text):
  warningFlags.push("⚠️ Enthält Fluorid")
  // KEIN Score-Abzug — nur Flag

// Zucker-Ausschluss im Profil
if user_profile.avoid_sugar:
  if sugar > 15:   penalty = 15
  elif sugar > 5:  penalty = 5
  else:            penalty = 0

  // Typ-Modifikator
  if sugarCtx.hasIndustrialSugar:
    penalty *= 1.0   // volle Strafe
  elif sugarCtx.hasNaturalSugar:
    penalty *= 0.6   // reduziert
  elif sugarCtx.hasFruitSugar:
    penalty *= 0.3   // mild
  // Mischfall: worst-case gewinnt

  // Kontext-Override (abmildern, nie freisprechen)
  if cacao > 70 and sugar < 15:
    penalty *= 0.5
    contextLabel = "Bewusstes Genussmittel"
  elif sugarCtx.hasFruitSugar and 
       not sugarCtx.hasIndustrialSugar:
    penalty *= 0.4
    contextLabel = "Natürlicher Fruchtzucker"

profile_score = cap(25 - penalty)

// Profile-Fit Label (KEINE verbotenen Wörter!)
if profile_score >= 20:
  label = "✅ Passt gut zu deinem Profil"
elif profile_score >= 10:
  label = "🟡 Mit Bedacht genießen"
else:
  label = "🔶 Für dein Profil nur 
           eingeschränkt passend"
